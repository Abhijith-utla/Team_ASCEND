def sol():
    def sat(li: List[int]):
        # Check if all elements in the list are either 2, 3, or 4 times their position
        return all(j in {i - 1, i + 1, 3 * i} for i, j in enumerate(li, start=1))

    answer = [2, 3, 4, 5, 6, 8, 12, 20]
    assert sat(answer)
    return answer

print(sol())
