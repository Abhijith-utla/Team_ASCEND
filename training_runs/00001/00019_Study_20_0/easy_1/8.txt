def sol():
    answer = []
    for i in range(1, 101):
        if i == 1 or i == 100:
            answer.append(i)
        elif i == 2 or i == 98:
            answer.append(i)
        elif i == 3 or i == 99:
            answer.append(i)
        else:
            answer.append(3 * i)
    return answer

print(sol())
