def sol():
    n = 20
    while n > 0:
        if n % 5 == 1:
            break
        n -= 1 + (n - 1) // 5
    return n > 0 and n % 5 == 1

# Checker
def sat(n: int):
    for i in range(5):
        assert n % 5 == 1
        n -= 1 + (n - 1) // 5
    return n > 0 and n % 5 == 1

# Run checker
assert sat(sol())
