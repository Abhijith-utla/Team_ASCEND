def sol():
    class Answer:
        def __init__(self, moves, source, target):
            self.moves = moves
            self.source = source
            self.target = target

    return Answer([], [[0, 7], [4, 5, 6], [1, 2, 3, 8]], [[0, 1, 2, 3], [4, 5], [6, 7]])

assert sat(sol().moves, sol().source, sol().target)
