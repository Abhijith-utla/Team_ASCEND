import random

def generate_coords(n):
    coords = []
    for i in range(n):
        x = random.randint(0, n-1)
        y = random.randint(0, n-1)
        coords.append((x, y))
    return coords

def is_valid(coords, side, num_points):
    for i1 in range(len(coords)):
        x1, y1 = coords[i1]
        assert 0 <= x1 < side and 0 <= y1 < side
        for i2 in range(i1):
            x2, y2 = coords[i2]
            for i3 in range(i2):
                x3, y3 = coords[i3]
                assert x1 * (y2 - y3) + x2 * (y3 - y1)
