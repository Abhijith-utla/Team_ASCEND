def sol():
    return {
        "answer": "a"*200 + "aa"*201 + "aaa"*198,
        "found_chars": {
            "a": 200,
            "aa": 201,
            "aaa": 198
        }
    }

# Testing the function
def test_sol():
    result = sol()
    assert result["answer"] == "a"*200 + "aa"*201 + "aaa"*198
    assert result["found_chars"] == {"a": 200, "aa": 201, "aaa": 198}

test_sol()
