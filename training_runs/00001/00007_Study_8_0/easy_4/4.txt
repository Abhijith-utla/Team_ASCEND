def sol(ls):
    for idx1 in range(len(ls)):
        for idx2 in range(idx1 + 1, len(ls)):
            if ls[idx1] in ls[idx2] and ls[idx1] != ls[idx2]:
                return ls[idx1]
    return None

assert sat(sol(['a', 'b', 'c', 'd', 'e']))
assert not sat(sol(['a', 'b', 'b', 'd', 'e']))
assert sat(sol(['a', 'b', 'c', 'd', 'e']))
assert not sat(sol(['a', 'a', 'c', 'd', 'e']))
