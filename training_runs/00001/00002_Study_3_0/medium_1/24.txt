def sol():
    def sat(li: List[int]):
        return sorted(li) == list(range(999)) and all(li[i] != i for i in range(len(li)))

    assert sat([100, 200, 300, 400, 500])
    assert not sat([100, 200, 300, 401, 500])

    assert not sat([4, 3, 2, 1, 0])
