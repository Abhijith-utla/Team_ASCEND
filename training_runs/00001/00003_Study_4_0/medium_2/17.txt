```python
from typing import List

def sat(li: List[int]) -> bool:
    if len(li) != 10:
        return False
    if li.count(li[3]) != 2:
        return False
    if li.index(li[3]) == 3:
        return False
    if li.count(li[3]) != 1:
        return False
    return True
```

This function is correct because it checks if the list `li` has exactly 10 elements and if the number 3 appears twice in the list and the third occurrence of 3 is at the third position.
