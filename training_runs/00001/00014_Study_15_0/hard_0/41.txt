```python
from typing import List

def sol() -> List[int]:
    result = []
    for i in range(20):
        if sum(result) + 2 ** i - 1 > 2 ** i:
            break
        result.append(2 ** i - 1)
    return result

    return all(sum(li[:i]) == 2 ** i - 1 for i in range(20))
```

The `sat` function checks whether the sum of the first `i` elements of the list `li` is equal to `2 ** i - 1` for `i` in the range from 0 to 19.
