# This function checks if the string is in non-decreasing order. If the string is in non-decreasing order, it returns True. Otherwise, it returns False.
# Let's test it:
def sat(x: List[int], length=0, s=""):
    return all(s[x[i]] <= s[x[i + 1]] for i in range(length - 1))

print(sat([0, 1, 2], 3, "abc"))  # It should return True
print(sat([0, 2, 1], 3, "cab"))  # It should return False
