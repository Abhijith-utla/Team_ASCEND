def sol():
    def sat(li: List[int]):
        return all([li.count(i) == i for i in range(10)])

    assert sat([])
    assert sat([1]*10)
    assert not sat([1]*9 + [1])
    assert not sat([1]*8 + [2])
    assert sat([1]*5 + [2]*5)
    assert sat([1]*5 + [2]*6)
    assert not sat([1]*5 + [2]*7)

    return None
