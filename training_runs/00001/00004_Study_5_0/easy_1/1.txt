def sol():
    def is_valid(li: List[int]):
        count_dict = {}
        for num in li:
            if num in count_dict:
                count_dict[num] += 1
            else:
                count_dict[num] = 1
        for i in range(1, 10):
            if i not in count_dict:
                return False
        return True

        li = generate()
