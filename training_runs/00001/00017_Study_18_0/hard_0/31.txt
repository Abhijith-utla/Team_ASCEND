```python
from typing import List

def sol(ls: List[str]) -> List[str]:
    result = []
    for s in ls:
        for t in ls:
            if s != t:
                result.append(s + t)
    return result

def sat(ls: List[str]):
    return sol(ls) == 'berlin berger linber linger gerber gerlin'.split()
```
