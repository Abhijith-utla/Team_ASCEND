```python
from typing import List

def sol(ls: List[str]) -> str:
    return ' '.join(s + t for s in ls for t in ls if s != t)

assert sol(['berlin', 'berger', 'linber', 'gerber', 'gerlin']) == 'berlin berger linber gerber gerlin'
```
