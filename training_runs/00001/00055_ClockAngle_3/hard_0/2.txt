def sol(hands: List[int], target_angle=138):
    h, m = hands
    assert 0 < h <= 12 and 0 <= m < 60
    hour_angle = 30 * h + m / 2
    minute_angle = 6 * m
    return abs(hour_angle - minute_angle) in [target_angle, 360 - target_angle]

print(sol([12, 0]))  # True
print(sol([12, 30]))  # True
print(sol([12, 31]))  # False
print(sol([1, 30]))  # False
print(sol([1, 0]))  # True
print(sol([1, 31]))  # False
print(sol([1, 45]))  # True
print(sol([1, 44]))  # False
