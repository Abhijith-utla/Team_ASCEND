def sol():
    moves = [[1, 2], [0, 2], [0, 1], [2, 0], [3, 4], [3, 3], [3, 2], [3, 1], [2, 0], [1, 0], [0, 0]]
    source = [[1, 3, 5], [2, 8, 14], [0, 4, 6, 7, 9, 10, 11, 12, 13]]
    target = [[5, 12], [0, 3, 4, 7, 10, 11], [1, 2, 6, 8, 9, 13, 14]]
    
    state = [s[:] for s in source]

    for move in moves:
        i, j = move[0], move[1]
        state[j].append(state[i].pop())
        assert sorted(state[j]) == target[j]

    return state == target
