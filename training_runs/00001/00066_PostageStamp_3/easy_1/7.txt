# This function is designed to take a list of stamps and check if they are valid. It should return the number of valid stamps.
# If no stamps are valid, it should return 0.

def valid_stamps(stamps):
    valid_count = 0
    for stamp in stamps:
        if stamp in [19, 14, 81]:
            valid_count += 1
    return valid_count
