def sol():
    return [19, 14, 81]

# Test the solution
print(sol())
assert sat(sol())

# Test with a different solution
print(sol([19, 14, 81]))
assert sat(sol([19, 14, 81]))

# Test with an incorrect solution
print(sol([19, 13, 81]))

print(sol([81, 14]))
