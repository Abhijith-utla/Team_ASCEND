```python
from typing import List

def sat(li: List[int]) -> bool:
    prefix_sum = [0] * len(li)
    prefix_sum[0] = li[0]
    for i in range(1, len(li)):
        prefix_sum[i] = prefix_sum[i-1] + li[i]
    for i in range(1, len(li)):
        if prefix_sum[i-1] >= prefix_sum[i]:
            return False
    return True
```
