def sol():
    li = [random.randint(1, 20) for _ in range(20)]
    while not sat(li):
        li = [random.randint(1, 20) for _ in range(20)]
    return li

def sat(li: List[int]):
    return all([i < sum(li[:i]) for i in range(20)])

if __name__ == "__main__":
    ans = sol()
    print(ans)
    assert sat(ans)
