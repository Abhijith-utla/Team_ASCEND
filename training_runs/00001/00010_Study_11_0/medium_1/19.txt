```python
from typing import List

def sat(ls: List[str]):
    if len(ls) == 0:
        return True
    else:
        return all(i in ls for i in range(len(ls)))

# Testing
print(sat([])) # True
print(sat(["0"])) # True
print(sat(["0", "1"])) # True
print(sat(["0", "2"])) # False
```
