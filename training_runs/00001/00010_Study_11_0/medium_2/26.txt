```python
from typing import List

def sol(ls: List[str]):
    if len(ls) == 0:
        return True
    else:
        return all(len(i) == len(ls[0]) for i in ls)

# Test cases
print(sol([]))  # True
print(sol(["abc", "def", "ghijk"]))  # False
print(sol(["abcdefg", "hij", "klmnop"]))  # True
```

In this solution, we first check if the list is empty, in which case we return True, as an empty list satisfies the condition. If the list is not empty, we then check if all strings in the list have the same length. If they do, we return True; otherwise, we return False.
