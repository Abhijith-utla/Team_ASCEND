```python
from typing import List

def sat(ls: List[str]):
    if len(ls) == 0:
        return True
    else:
        return all(len(set(i)) == len(ls[0]) for i in ls)

def sol():
    return sat(['abcd', 'efgh', 'ijkl'])
```

The `sol` function calls `sat` function with a list of strings.
