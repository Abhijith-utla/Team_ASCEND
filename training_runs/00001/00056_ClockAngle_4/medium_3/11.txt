def sol():
    for hour in range(12):
        for minute in range(60):
            if sat(hour, minute):
                return hour, minute

# Checker
def check_answer(answer):
    hour, minute = answer
    hour_angle = 30 * hour + minute / 2
    minute_angle = 6 * minute
    assert sat(hour, minute) or (abs(hour_angle - minute_angle) in [180, 360 - 180])

answer = sol()
check_answer(answer)
