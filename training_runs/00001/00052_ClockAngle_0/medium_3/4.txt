# Test cases
assert sat(sol())
assert not sat((11, 59))
assert not sat((12, 0))
assert sat((12, 30))
assert not sat((12, 60))
assert not sat((23, 0))
assert not sat((0, 0))

# Checker
def sat(hands: Tuple[int, int]) -> bool:
    h, m = hands
    assert 0 <= h < 12 and 0 <= m < 60
    hour_angle = 30 * h + m / 12
    minute_angle = 6 * m
    return abs(hour_angle - minute_angle) in [target_angle, 360 - target_angle]
