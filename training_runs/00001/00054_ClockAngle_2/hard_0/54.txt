def sol():
    return {}, 0

def sol(hands: List[int], target_angle=133):
    h, m = hands
    assert 0 < h <= 12 and 0 <= m < 60
    hour_angle = 30 * h + m / 2
    minute_angle = 6 * m
    return {"angle_difference": abs(hour_angle - minute_angle) in [target_angle, 360 - target_angle]}

print(sol())
print(sol((5, 30)))
print(sol((12, 0)))
print(sol((11, 59)))
print(sol((10, 30)))
