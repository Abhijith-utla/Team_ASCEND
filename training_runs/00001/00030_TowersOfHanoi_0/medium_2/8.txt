def sol():
    moves = [(3, 5), (0, 7), (1, 2), (3, 1), (2, 6), (1, 0), (4, 7), (5, 4), (0, 2), (2, 1), (5, 6), (4, 5)]
    rods = ([], [], [8, 7, 6, 5, 4, 3, 2, 1], 10)
    for i, j in moves:
        rods[j].append(rods[i].pop())
        assert rods[j][-1] == min(rods[j]), "larger disk on top of smaller disk"
    assert rods[0] == [1, 2, 3, 4, 5, 6, 7, 8], "Rod 0 doesn't have the correct configuration"
    return rods[0]

print(sol())
