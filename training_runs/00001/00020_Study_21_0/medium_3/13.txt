def sol():
    def sat(li: List[int]):
        return all([li[i] != li[i + 1] for i in range(7)]) and len(set(li)) == 3

    assert sat([1, 2, 2, 3, 4, 4, 4])
    assert not sat([1, 2, 2, 3, 3, 4, 4])
    assert not sat([1, 1, 2, 2, 3, 3, 4])
    assert sat([1, 2, 3, 4, 5, 6, 7])
    assert not sat([1, 2, 2, 3, 3, 3, 4])
    assert not sat([1, 2, 2, 2, 3, 3, 4])
    assert not sat([1, 2, 3, 4, 4, 4, 5])
    assert sat([1, 1, 1, 2, 2, 2, 3])
