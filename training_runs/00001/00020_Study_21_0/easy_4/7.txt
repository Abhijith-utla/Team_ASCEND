```python
def sol():
    li = [1, 2, 3, 4, 5, 6, 7, 8]
    return sat(li)

def sat(li: List[int]):
    return all([li[i] != li[i + 1] for i in range(8)] and len(set(li)) == 3)

print(sat(sol()))
```

The function returns `True` if the list has no two consecutive elements and all elements are unique.
