def sol():
    def sat(li: List[int]):
        return all([li[i] != li[i + 1] for i in range(8)]) and len(set(li)) == 3

    assert sat([1, 2, 3, 4, 4, 5, 6, 7])
    assert not sat([1, 2, 3, 4, 4, 5, 6, 7, 7])
    assert not sat([1, 1, 1, 1, 1, 1, 1, 1])
    assert sat([1, 1, 1, 1, 2, 2, 2, 2])
    print("All tests passed")

sol()
